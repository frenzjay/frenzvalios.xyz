Facebook Downloader API Documentation - Ruri

API Endpoint

POST /

Request Parameters

  • type (optional): The type of download (default/raw)
  • url (required): The URL of the Facebook video, live stream, or story

Examples

Using curl

curl -X POST -F "type=default" -F "url=your_url_here" https://fbdl.ruriyoshinova.dev/

Using Python

import requests

url = "https://fbdl.ruriyoshinova.dev/"

payload = {
    "type": "default",
    "url": "your_url_here"
}

response = requests.post(url, data=payload)

if response.status_code == 200:
    download_link = response.text
    print("Download Link:", download_link)
else:
    print("Error:", response.status_code)

Using Node.js

const axios = require('axios');

const url = "https://fbdl.ruriyoshinova.dev/";

const payload = new URLSearchParams();
payload.append('type', 'default');
payload.append('url', 'your_url_here');

axios.post(url, payload)
    .then(response => {
        if (response.status === 200) {
            const downloadLink = response.data;
            console.log("Download Link:", downloadLink);
        } else {
            console.log("Error:", response.status);
        }
    })
    .catch(error => {
        console.error("Error:", error);
    });
